# Payload CMS Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Integrate Payload CMS 3.x into the existing Next.js 16.2.2 clinic website, replacing hardcoded data with CMS-managed content and building a full News/Blog system with block-based editing.

**Architecture:** Payload runs embedded inside the same Next.js app. Admin panel at `/admin`, website at all other routes. SQLite database (file-based). Existing client components are modified to accept data via props from server-component parents that fetch from Payload's Local API. A data access layer (`lib/payload-data.ts`) maps Payload documents to existing TypeScript interfaces for backward compatibility.

**Tech Stack:** Payload CMS 3.83+, @payloadcms/db-sqlite, @payloadcms/richtext-lexical, sharp, graphql

**Spec:** `docs/superpowers/specs/2026-04-18-payload-cms-design.md`

Create `src/payload.config.ts`:

```ts
import { buildConfig } from 'payload'
import { sqliteAdapter } from '@payloadcms/db-sqlite'
import { lexicalEditor } from '@payloadcms/richtext-lexical'
import path from 'path'
import { fileURLToPath } from 'url'
import sharp from 'sharp'

import { Media } from './collections/Media'
import { Users } from './collections/Users'
import { Services } from './collections/Services'
import { Doctors } from './collections/Doctors'
import { CheckupPackages } from './collections/CheckupPackages'
import { Reviews } from './collections/Reviews'
import { News } from './collections/News'
import { SiteSettings } from './globals/SiteSettings'

const filename = fileURLToPath(import.meta.url)
const dirname = path.dirname(filename)

export default buildConfig({
  secret: process.env.PAYLOAD_SECRET || 'default-secret-change-me',

  db: sqliteAdapter({
    client: {
      url: process.env.DATABASE_URL || 'file:./database.db',
    },
    push: true,
  }),

  editor: lexicalEditor({}),

  collections: [Media, Users, Services, Doctors, CheckupPackages, Reviews, News],

  globals: [SiteSettings],

  localization: {
    locales: [
      { label: 'ქართული', code: 'ge' },
      { label: 'English', code: 'en' },
      { label: 'Русский', code: 'ru' },
    ],
    defaultLocale: 'ge',
    fallback: true,
  },

  admin: {
    user: 'users',
    importMap: {
      baseDir: path.resolve(dirname),
    },
    meta: {
      titleSuffix: ' — Khozrevanidze Clinic',
    },
  },

  typescript: {
    autoGenerate: true,
    outputFile: path.resolve(dirname, 'payload-types.ts'),
  },

  sharp,
  telemetry: false,
})
```

> **Note:** This file references collections and globals that will be created in subsequent tasks. The app will not compile until all collections are created. That is expected.

import config from '@payload-config'

// Map Payload locale codes to our locale codes (they match, but this makes it explicit)
type Locale = 'ge' | 'en' | 'ru'

async function getPayloadClient() {
  return getPayload({ config })

export async function getServices(locale: Locale): Promise<Service[]> {
  const payload = await getPayloadClient()
  const result = await payload.find({
    collection: 'services',
    locale,
    limit: 100,
    depth: 1,
  })

  return result.docs.map((doc) => ({
    id: String(doc.id),
    slug: doc.slug,
    name: doc.name,
    description: doc.description,
    shortDescription: doc.shortDescription,
    icon: doc.icon,
    image: typeof doc.image === 'object' && doc.image !== null ? doc.image.url ?? '' : '',
  }))
}

export async function getDoctors(locale: Locale, limit?: number): Promise<Doctor[]> {
  const payload = await getPayloadClient()
  const result = await payload.find({
    collection: 'doctors',
    locale,
    limit: limit ?? 100,
    depth: 1,
  })

  return result.docs.map((doc) => ({
    id: String(doc.id),
    slug: doc.slug,
    name: doc.name,
    specialty: doc.specialty,
    photo: typeof doc.photo === 'object' && doc.photo !== null ? doc.photo.url ?? '' : '',
    biography: '',
    qualifications: (doc.qualifications ?? []).map((q: { qualification: string }) => q.qualification),
    specializations: (doc.specializations ?? []).map((s: { specialization: string }) => s.specialization),
    experienceYears: doc.experienceYears ?? 0,
    languagesSpoken: (doc.languagesSpoken ?? []).map((l: { language: string }) => l.language),
    isDepartmentHead: doc.isDepartmentHead ?? false,
  }))
}

export async function getCheckupPackages(locale: Locale): Promise<CheckupPackage[]> {
  const payload = await getPayloadClient()
  const result = await payload.find({
    collection: 'checkup-packages',
    locale,

    depth: 0,
  })

  return result.docs.map((doc) => ({
    id: String(doc.id),
    name: doc.name,
    description: doc.description,
    price: doc.price,
    currency: doc.currency,
    duration: doc.duration,
    includedServices: (doc.includedServices ?? []).map((s: { service: string }) => s.service),
    isFeatured: doc.isFeatured ?? false,
  }))
}

export async function getReviews(locale: Locale): Promise<Review[]> {

  const result = await payload.find({
    collection: 'reviews',
    locale,
    limit: 100,
    depth: 0,
    sort: '-date',
  })

  return result.docs.map((doc) => ({
    id: String(doc.id),
    author: doc.author,
    rating: doc.rating,
    text: doc.text,
    date: doc.date,

  }))
}

export async function getStats() {
  const payload = await getPayloadClient()
  const settings = await payload.findGlobal({
    slug: 'site-settings',
    depth: 0,
  })

  return {
    patients: settings.stats?.patients ?? 15000,
    doctors: settings.stats?.doctors ?? 54,
    operations: settings.stats?.operations ?? 5000,
    experience: settings.stats?.experience ?? 9,
  }
}

export async function getHomepageNews(locale: Locale) {
  const payload = await getPayloadClient()
  const result = await payload.find({
    collection: 'news',
    locale,
    depth: 1,
    where: {
      showOnHomepage: { equals: true },
      status: { equals: 'published' },
    },
    sort: 'homepageOrder',
    limit: 6,
  })

  return result.docs.map((doc) => ({
    id: String(doc.id),
    slug: doc.slug,
    title: doc.title,
    excerpt: doc.excerpt,
    category: doc.category,
    publishedDate: doc.publishedDate,
    featuredImage: typeof doc.featuredImage === 'object' && doc.featuredImage !== null
      ? { url: doc.featuredImage.url ?? '', alt: doc.featuredImage.alt ?? '' }
      : { url: '', alt: '' },
  }))
}

export async function getAllNews(locale: Locale, page = 1, limit = 12) {
  const payload = await getPayloadClient()
  const result = await payload.find({
    collection: 'news',
    locale,
    depth: 1,
    where: {
      status: { equals: 'published' },
    },
    sort: '-publishedDate',
    page,
    limit,
  })

  return {
    docs: result.docs.map((doc) => ({
      id: String(doc.id),
      slug: doc.slug,
      title: doc.title,
      excerpt: doc.excerpt,
      category: doc.category,
      publishedDate: doc.publishedDate,
      featuredImage: typeof doc.featuredImage === 'object' && doc.featuredImage !== null
        ? { url: doc.featuredImage.url ?? '', alt: doc.featuredImage.alt ?? '' }
        : { url: '', alt: '' },
    })),
    totalPages: result.totalPages,
    page: result.page,
  }
}

export async function getNewsBySlug(slug: string, locale: Locale) {

  const result = await payload.find({
    collection: 'news',

    where: {

  const imgUrl = typeof image === "object" ? image.url : "";
  const imgAlt = typeof image === "object" ? image.alt : "";

  const alignClasses: Record<string, string> = {
    left: "float-left mr-8 mb-4 max-w-md",
    center: "mx-auto max-w-3xl",
    right: "float-right ml-8 mb-4 max-w-md",
    fullWidth: "w-full",
  };

  return (
    <figure className={`my-8 ${alignClasses[alignment] || alignClasses.center}`}>
      {imgUrl && (
        <img
          src={imgUrl}
          alt={imgAlt || ""}
          className="rounded-2xl w-full"
          loading="lazy"
        />
      )}
      {caption && (
        <figcaption className="text-[13px] text-grey-light text-center mt-3 italic">
          {caption}
        </figcaption>
      )}
    </figure>
  );
}

function ImageTextBlockRenderer({
  image,
  imagePosition,
  content,
}: {
  image: { url?: string; alt?: string } | number;
  imagePosition: string;
  content: SerializedEditorState;
}) {
  const imgUrl = typeof image === "object" ? image.url : "";
  const imgAlt = typeof image === "object" ? image.alt : "";

  return (
    <div
      className={`my-10 flex flex-col md:flex-row gap-8 items-center ${
        imagePosition === "right" ? "md:flex-row-reverse" : ""
      }`}
    >
      <div className="md:w-1/2">
        {imgUrl && (
          <img
            src={imgUrl}
            alt={imgAlt || ""}
            className="rounded-2xl w-full"
            loading="lazy"
          />
        )}
      </div>
      <div className="md:w-1/2 prose prose-lg max-w-none prose-headings:text-blackberry prose-p:text-grey prose-a:text-pink">
        <RichText data={content} />
      </div>
    </div>
  );
}

function QuoteBlockRenderer({
  quoteText,
  attribution,
}: {
  quoteText: string;
  attribution?: string;
}) {
  return (
    <blockquote className="my-10 border-l-4 border-pink pl-6 py-4 bg-pink-light/30 rounded-r-2xl">
      <p className="text-[18px] text-blackberry italic leading-relaxed">
        &ldquo;{quoteText}&rdquo;
      </p>
      {attribution && (
        <cite className="block mt-3 text-[14px] text-grey-light not-italic font-medium">
          — {attribution}
        </cite>
      )}
    </blockquote>
  );
}

export default function ContentBlocks({ blocks }: { blocks: Block[] }) {
  return (
    <div className="space-y-8 clearfix">
      {blocks.map((block, index) => {
        switch (block.blockType) {
          case "richText":
            return <RichTextBlockRenderer key={index} content={block.content} />;
          case "image":
            return (
              <ImageBlockRenderer
                key={index}
                image={block.image}
                alignment={block.alignment}
                caption={block.caption}
              />
            );
          case "imageText":
            return (
              <ImageTextBlockRenderer
                key={index}
                image={block.image}
                imagePosition={block.imagePosition}
                content={block.content}
              />
            );
          case "quote":
            return (
              <QuoteBlockRenderer
                key={index}
                quoteText={block.quoteText}
                attribution={block.attribution}
              />
            );
          default:
            return null;
        }
      })}
    </div>
  );
}

- [ ] **Step 1: Create the seed script**

Create `src/seed.ts`:

```ts
import { getPayload } from 'payload'
import config from '@payload-config'
import { services, doctors, checkupPackages, reviews, stats } from './lib/data'

async function seed() {
  const payload = await getPayload({ config })

  console.log('Seeding database...')

  // Check if already seeded
  const existingServices = await payload.find({ collection: 'services', limit: 1 })
  if (existingServices.totalDocs > 0) {
    console.log('Database already seeded. Skipping.')
    process.exit(0)
  }

  // Seed services
  console.log('Seeding services...')
  for (const service of services) {
    await payload.create({
      collection: 'services',
      data: {
        name: service.name,
        slug: service.slug,
        description: service.description,
        shortDescription: service.shortDescription,
        icon: service.icon as 'heart' | 'brain' | 'baby' | 'brain-circuit' | 'flask' | 'ear' | 'scissors' | 'activity',
      },
      locale: 'ge',
    })
  }
  console.log(`  Seeded ${services.length} services`)

  // Seed doctors
  console.log('Seeding doctors...')
  for (const doctor of doctors) {
    await payload.create({
      collection: 'doctors',
      data: {
        name: doctor.name,
        slug: doctor.slug,
        specialty: doctor.specialty,
        biography: undefined,
        experienceYears: doctor.experienceYears,
        isDepartmentHead: doctor.isDepartmentHead,
        qualifications: doctor.qualifications.map((q) => ({ qualification: q })),
        specializations: doctor.specializations.map((s) => ({ specialization: s })),
        languagesSpoken: doctor.languagesSpoken.map((l) => ({ language: l })),
      },
      locale: 'ge',
    })
  }
  console.log(`  Seeded ${doctors.length} doctors`)

  // Seed checkup packages
  console.log('Seeding checkup packages...')
  for (const pkg of checkupPackages) {
    await payload.create({
      collection: 'checkup-packages',
      data: {
        name: pkg.name,
        description: pkg.description,
        price: pkg.price,
        currency: pkg.currency,
        duration: pkg.duration,
        includedServices: pkg.includedServices.map((s) => ({ service: s })),
        isFeatured: pkg.isFeatured,
      },
      locale: 'ge',
    })
  }
  console.log(`  Seeded ${checkupPackages.length} checkup packages`)

  // Seed reviews
  console.log('Seeding reviews...')
  for (const review of reviews) {
    await payload.create({
      collection: 'reviews',
      data: {
        author: review.author,
        rating: review.rating,
        text: review.text,
        date: review.date,
        source: review.source as 'google' | 'internal',
      },
      locale: 'ge',
    })
  }
  console.log(`  Seeded ${reviews.length} reviews`)

  // Seed site settings
  console.log('Seeding site settings...')
  await payload.updateGlobal({
    slug: 'site-settings',
    data: {
      stats: {
        patients: stats.patients,
        doctors: stats.doctors,
        operations: stats.operations,
        experience: stats.experience,
      },
    },
  })
  console.log('  Seeded site settings')

  // Create admin user if none exists
  const existingUsers = await payload.find({ collection: 'users', limit: 1 })
  if (existingUsers.totalDocs === 0) {
    console.log('Creating admin user...')
    await payload.create({
      collection: 'users',
      data: {
        email: 'admin@khozrevanidze.ge',
        password: 'ChangeMe123!',
        role: 'admin',
        name: 'Admin',
      },
    })
    console.log('  Admin user created: admin@khozrevanidze.ge / ChangeMe123!')
  }

  console.log('\nSeeding complete!')
  process.exit(0)
}

seed().catch((err) => {
  console.error('Seed error:', err)
  process.exit(1)
})
```
